Sửa lỗi JavaScript

Đoạn mã lỗi:
function deleteRow(index) { 
  let confirm = confirm("Bạn có chắc muốn xóa?"); 
  if (confirm) { 
    listData.splice(index, 0); 
    renderTable(); 
  } 
}
Đoạn mã đã sửa:
function deleteRow(index) {

    // SỬA LỖI 1:
    // Không đặt tên biến là "confirm"
    // vì trùng với hàm confirm() của JavaScript

    let isConfirm = confirm("Bạn có chắc muốn xóa?");

    if (isConfirm) {
// SỬA LỖI 2:
        // splice(index, 0) là không xóa phần tử nào
        // phải sửa thành splice(index, 1)

        listData.splice(index, 1);

        renderTable();
    }
}
